209    def result_exists(self, context_length, depth_percent):
210        """
211        Checks to see if a result has already been evaluated or not
212        """
213
214        results_dir = 'results/'
215        if not os.path.exists(results_dir):
216            return False
217        
218        for filename in os.listdir(results_dir):
219            if filename.endswith('.json'):
220                with open(os.path.join(results_dir, filename), 'r') as f:
221                    result = json.load(f)
222                    context_length_met = result['context_length'] == context_length
223                    depth_percent_met = result['depth_percent'] == depth_percent
224                    version_met = result.get('version', 1) == self.results_version
225                    model_met = result['model'] == self.model_name
226                    if context_length_met and depth_percent_met and version_met and model_met:
227                        return True
228        return False

244    def insert_needle(self, context, depth_percent, context_length):
245        tokens_needle = self.model_to_test.encode_text_to_tokens(self.needle)
246        tokens_context = self.model_to_test.encode_text_to_tokens(context)
247
248        # Reducing the context length by 150 buffer. This is to account for system message, the user question, and response.
249        context_length -= self.final_context_length_buffer
250
251        # If your context + needle are longer than the context length (which it will be), then reduce tokens from the context by the needle length
252        if len(tokens_context) + len(tokens_needle) > context_length:
253            tokens_context = tokens_context[:context_length - len(tokens_needle)]
254
255        if depth_percent == 100:
256            # If your depth percent is 100 (which means your needle is the last thing in the doc), throw it at the end
257            tokens_new_context = tokens_context + tokens_needle
258        else:
259            # Go get the position (in terms of tokens) to insert your needle
260            insertion_point = int(len(tokens_context) * (depth_percent / 100))
261
262            # tokens_new_context represents the tokens before the needle
263            tokens_new_context = tokens_context[:insertion_point]
264
265            # We want to make sure that we place our needle at a sentence break so we first see what token a '.' is
266            period_tokens = self.model_to_test.encode_text_to_tokens('.')
267            
268            # Then we iteration backwards until we find the first period
269            while tokens_new_context and tokens_new_context[-1] not in period_tokens:
270                insertion_point -= 1
271                tokens_new_context = tokens_context[:insertion_point]
272
273            # Once we get there, then add in your needle, and stick the rest of your context in on the other end.
274            # Now we have a needle in a haystack
275            tokens_new_context += tokens_needle + tokens_context[insertion_point:]
276
277        # Convert back to a string and return it
278        new_context = self.model_to_test.decode_tokens(tokens_new_context)
279        return new_context

281    def get_context_length_in_tokens(self, context):
282        return len(self.model_to_test.encode_text_to_tokens(context))

284    def read_context_files(self):
285        context = ""
286        max_context_length = max(self.context_lengths)
287        base_dir = os.path.abspath(os.path.dirname(__file__))  # Package directory
288
289        while self.get_context_length_in_tokens(context) < max_context_length:
290            for file in glob.glob(os.path.join(base_dir, self.haystack_dir, "*.txt")):
291                with open(file, 'r') as f:
292                    context += f.read()
293        return context

295    def encode_and_trim(self, context, context_length):
296        tokens = self.model_to_test.encode_text_to_tokens(context)
297        if len(tokens) > context_length:
298            context = self.model_to_test.decode_tokens(tokens, context_length)
299        return context
